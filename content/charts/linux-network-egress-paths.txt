        APP1["<b>Normal socket</b><br/>socket(AF_INET, SOCK_STREAM/DGRAM)<br/>send() / write()<br/><br/>user: payload<br/>kernel builds: L2 + L3 + L4"]
        APP2A["<b>Raw IP, kernel hdr</b><br/>socket(AF_INET, SOCK_RAW, proto)<br/>no IP_HDRINCL<br/>CAP_NET_RAW<br/><br/>user: L4 hdr + payload<br/>kernel builds: L2 + L3"]
        APP2B["<b>Raw IP, user hdr</b><br/>socket(AF_INET, SOCK_RAW, proto)<br/>setsockopt IP_HDRINCL<br/>CAP_NET_RAW<br/><br/>user: L3 + L4 + payload<br/>kernel builds: L2 and may finalize some IP fields"]
        APP3["<b>AF_PACKET SOCK_DGRAM</b><br/>socket(AF_PACKET, SOCK_DGRAM)<br/>CAP_NET_RAW<br/><br/>user: L3 + L4 + payload<br/>kernel builds: L2 (from sockaddr_ll)"]
        APP4["<b>AF_PACKET SOCK_RAW</b><br/>socket(AF_PACKET, SOCK_RAW)<br/>CAP_NET_RAW<br/><br/>user: L2 + L3 + L4 + payload<br/>kernel builds: nothing"]
        APP5["<b>Userspace driver</b><br/>DPDK / VFIO / AF_XDP zerocopy<br/>CAP_SYS_RAWIO + vfio-pci<br/><br/>user: entire frame + NIC DMA<br/>kernel: not involved"]
    end
